60       public View getView(final int position, View view, ViewGroup parent) { 
61           final ViewHolder holder; 
62           if (view == null) { 
63               holder = new ViewHolder(); 
64               view = inflater.inflate(R.layout.listview_item, null); 
65               // Locate the TextViews in listview_item.xml 
66               holder.D_Id = (TextView) view.findViewById(R.id.Doctor_id); 
67               holder.D_Name = (TextView) view.findViewById(R.id.Doctor_name); 
68               holder.population = (TextView) view.findViewById(R.id.Appointment); 
69               holder.P_Appointment_No = (TextView) view.findViewById(R.id.Appointment_no); 
70               view.setTag(holder); 
71           } else { 
72               holder = (ViewHolder) view.getTag(); 
73           } 
74           // Set the results into TextViews 
75           holder.D_Id.setText(HeathfieldAppointment.get(position).getDoctor_id()); 
76           holder.D_Name.setText(HeathfieldAppointment.get(position).getDoctor_Name()); 
77           holder.population.setText(HeathfieldAppointment.get(position).getAppointment_Date()); 
78           holder.P_Appointment_No.setText((HeathfieldAppointment.get(position).getAppointment_number()));

90                           (HeathfieldAppointment.get(position).getDoctor_id())); 
91                   // Pass all data D_Name 
92                   intent.putExtra("D_Name", 
93                           (HeathfieldAppointment.get(position).getDoctor_Name())); 
94                   // Pass all data P_Appointment 
95                   intent.putExtra("P_Appointment", 
96                           (HeathfieldAppointment.get(position).getAppointment_Date())); 
97                   intent.putExtra("P_Appointment_No", 
98                           (HeathfieldAppointment.get(position).getAppointment_number())); 
99                   intent.putExtra("Cancel","CancelAppointment"); 
100                  intent.putExtra("Book","CancelAppointment"); 
101   
102   
103                  // Start Booking_Appointments_View Class

107              }
