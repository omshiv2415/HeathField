20    
21   public class Appointment_Confirmation_View extends Activity { 
22       // Declare Variables 
23       TextView PatientId; 
24       TextView PatientAppointmentdate; 
25       TextView DoctorName; 
26       TextView DoctorID; 
27    
28       String P_ID; 
29       String P_APPOINTMENT; 
30       String P_DOCTOR_NAME; 
31       String P_DoctorId; 
32    
33       Button ConfirmAppointment; 
34    
35       @Override 
36       public void onCreate(Bundle savedInstanceState) { 
37           super.onCreate(savedInstanceState); 
38           setContentView(R.layout.confirm_singleitemview); 
39           // Retrieve data from booking appointment on item click event 
40           Intent i = getIntent(); 
41    
42           P_ID = i.getStringExtra("PatientId"); 
43           P_APPOINTMENT = i.getStringExtra("AppointmentDate"); 
44           P_DOCTOR_NAME = i.getStringExtra("DoctorName"); 
45           P_DoctorId = i.getStringExtra("DoctorID"); 
46    
47    
48           // Locate the TextViews in singleitemview.xml 
49    
50           PatientAppointmentdate = (TextView) findViewById(R.id.Appointment); 
51           DoctorName = (TextView) findViewById(R.id.Doctor_name); 
52           DoctorID = (TextView)findViewById(R.id.Doctor_id); 
53           PatientId = (TextView)findViewById(R.id.Appointment_no); 
54    
55    
56           ConfirmAppointment = (Button)findViewById(R.id.appointment); 
57    
58           // Load the results into the TextViews 
59    
60           PatientAppointmentdate.setText(P_APPOINTMENT); 
61           DoctorName.setText(P_DOCTOR_NAME); 
62           DoctorID.setText(P_DoctorId); 
63           PatientId.setText(P_ID); 
64    
65           ConfirmAppointment.setOnClickListener(new View.OnClickListener() { 
66    
67               String pid = PatientId.getText().toString(); 
68               String pappointment = PatientAppointmentdate.getText().toString(); 
69               String pdocotor = DoctorName.getText().toString().trim(); 
70               String pdoctorId = DoctorID.getText().toString().trim(); 
71    
72    
73               @Override 
74               public void onClick(View view) { 
75    
76    
77                   ParseObject PatientPrescription = new ParseObject("PatientAppointment"); 
78    
79    
80                   PatientPrescription.put("createdBy", pid); 
81                   PatientPrescription.put("MyAppointment",pappointment); 
82                   PatientPrescription.put("DoctorName",pdocotor); 
83                   PatientPrescription.put("DrID",pdoctorId); 
84                   PatientPrescription.put("Appointment_No","No Number (Requested)"); 
85    
86    
87                       PatientPrescription.saveInBackground(); 
88                      // sending a notification to user once they booked the appointment 
89                       String check = new ParseUser().getObjectId(); 
90    
91                       ParseQuery<ParseInstallation> pQuery = ParseInstallation.getQuery(); 
92                       pQuery.whereEqualTo( pid,check ); 
93    
94    
95    
96                       ParsePush push = new ParsePush(); 
97                       push.setQuery(pQuery); 
98                       push.setMessage("Your Appointment Request has been Approved for "+ pappointment); 
99                       push.sendInBackground(); 
100                      Toast toast = Toast.makeText(Appointment_Confirmation_View.this, "Appointment Confirmed Successfully ", Toast.LENGTH_LONG); 
101                      toast.setGravity(Gravity.CENTER, 0, 0); 
102                      toast.show(); 
103                      Intent takeUserHome = new Intent(Appointment_Confirmation_View.this, Doctor.class); 
104                      startActivity(takeUserHome); 
105                      Appointment_Confirmation_View.this.finish(); 
106   
107              } 
108          });
