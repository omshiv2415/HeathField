288      // validating email id 
289      private boolean isValidEmail(String email) { 
290          String EMAIL_PATTERN = "^[_A-Za-z0-9-\\+]+(\\.[_A-Za-z0-9-]+)*@" 
291                  + "[A-Za-z0-9-]+(\\.[A-Za-z0-9]+)*(\\.[A-Za-z]{2,})$"; 
292   
293          Pattern pattern = Pattern.compile(EMAIL_PATTERN); 
294          Matcher matcher = pattern.matcher(email); 
295          return matcher.matches(); 
296      }
