183      // RemoteDataTask AsyncTask 
184      private class RemoteDataTask extends AsyncTask<Void, Void, Void> { 
185          @Override 
186          protected void onPreExecute() { 
187              super.onPreExecute(); 
188              // Create a progressdialog 
189              mProgressDialog = new ProgressDialog(BookingAppointment.this); 
190              // Set progressdialog title 
191              mProgressDialog.setTitle("HeathField Appointment Booking"); 
192              // Set progressdialog message 
193              mProgressDialog.setMessage("Loading..."); 
194              mProgressDialog.setIndeterminate(false); 
195              // Show progressdialog 
196              mProgressDialog.show(); 
197          } 
198   
199          @Override 
200          protected Void doInBackground(Void... params) { 
201              // Create the array 
202              HeathFieldSurgery = new ArrayList<App_Booking_and_cancel_support>(); 
203              try { 
204                  // Locate the class table named "Country" in Parse.com 
205                  ParseQuery<ParseObject> query = new ParseQuery<ParseObject>( 
206                          "Doctor"); 
207                  // Locate the column named "ranknum" in Parse.com and order list 
208                  // by ascending 
209                  query.orderByAscending("Appointment_Number"); 
210                  ob = query.find(); 
211                  for (ParseObject country : ob) { 
212                      App_Booking_and_cancel_support map = new App_Booking_and_cancel_support(); 
213                      map.setDoctor_id((String) country.get("Doctor_ID")); 
214                      map.setDoctor_Name((String) country.get("Doctor_Name")); 
215                      map.setAppointment_Date((String) country.get("Appointment_Slot").toString()); 
216                      map.setAppointment_number((String) country.get("Appointment_Number")); 
217                      HeathFieldSurgery.add(map); 
218                  } 
219              } catch (ParseException e) { 
220                  Log.e("Error", e.getMessage()); 
221                  e.printStackTrace(); 
222              } 
223              return null; 
224          } 
225   
226   
227   
228          @Override 
229          protected void onPostExecute(Void result) { 
230              // Locate the listview in activity_booking_appointment.xmlappointment.xml 
231              listview = (ListView) findViewById(R.id.listview); 
232              // Pass the results into Booking_ListViewAdapter.java 
233              adapter = new Booking_ListViewAdapter(BookingAppointment.this, 
234                      HeathFieldSurgery); 
235              // Binds the Adapter to the ListView 
236              listview.setAdapter(adapter); 
237              // Close the progressdialog 
238              mProgressDialog.dismiss(); 
239   
240   
241          }

244      }
