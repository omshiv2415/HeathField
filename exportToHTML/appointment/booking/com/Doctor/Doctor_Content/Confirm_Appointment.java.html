53       private class RemoteDataTask extends AsyncTask<Void, Void, Void> { 
54           @Override 
55           protected void onPreExecute() { 
56               super.onPreExecute(); 
57               // Create a progressdialog 
58               mProgressDialog = new ProgressDialog(Confirm_Appointment.this); 
59               // Set progressdialog title 
60               mProgressDialog.setTitle("Prescription"); 
61               // Set progressdialog message 
62               mProgressDialog.setMessage("Loading..."); 
63               mProgressDialog.setIndeterminate(false); 
64               // Show progressdialog 
65               mProgressDialog.show(); 
66           } 
67    
68           @Override 
69           protected Void doInBackground(Void... params) { 
70               // Create the array 
71               AppointmentConfirmList = new ArrayList<Doctor_ConfirmAppointment_support>(); 
72               try { 
73                   // Locate the class table named "Country" in Parse.com 
74                   ParseQuery<ParseObject> query = new ParseQuery<ParseObject>( 
75                           "PatientAppointmentRequest"); 
76                   // query.whereEqualTo("createdBy",  ParseUser.getCurrentUser()); 
77    
78                   query.whereEqualTo("DrName", ParseUser.getCurrentUser().get("Name")); 
79    
80    
81    
82                   ob = query.find(); 
83                   for (ParseObject Prescription_Items : ob) { 
84                       Doctor_ConfirmAppointment_support map = new Doctor_ConfirmAppointment_support(); 
85                       map.setPatient_Id((String) Prescription_Items.get("RequestedId")); 
86                       map.setPatient_Name((String) Prescription_Items.get("PatientName")); 
87                       map.setDate_of_Birth((String) Prescription_Items.get("PatientDob")); 
88                       map.setAppointment_Date((String) Prescription_Items.get("RequestedAppointment")); 
89                       map.setDoctorname((String) Prescription_Items.get("DrName")); 
90                       map.setDocId((String) Prescription_Items.get("DoctorID")); 
91                       map.setPatientEmail((String) Prescription_Items.get("RequestedEmail")); 
92                       map.setPatientContactNo((String) Prescription_Items.get("PatientPhone")); 
93                       AppointmentConfirmList.add(map); 
94                   } 
95               } catch (ParseException e) { 
96                   Log.e("Error", e.getMessage()); 
97                   e.printStackTrace(); 
98               } 
99               return null; 
100          } 
101   
102          @Override 
103          protected void onPostExecute(Void result) { 
104              TextView noApp = (TextView)findViewById(R.id.noAppointments); 
105              // Locate the listview in activity_booking_appointmenting_appointment.xml 
106              listview = (ListView) findViewById(R.id.ConfirmListView); 
107              // Pass the results into Booking_ListViewAdapter.java 
108              adapter = new Doctor_ConfirmAppointment_ListViewAdapter(Confirm_Appointment.this, 
109                      AppointmentConfirmList); 
110              // Binds the Adapter to the ListView 
111              if(adapter.getCount()!=0){ 
112                  listview.setAdapter(adapter); 
113                  noApp.setVisibility(View.GONE); 
114              }else{ 
115                  Toast.makeText(Confirm_Appointment.this, "No Appointments Requested", Toast.LENGTH_SHORT).show(); 
116                  noApp.setVisibility(View.VISIBLE); 
117   
118              } 
119              // Close the progressdialog 
120              mProgressDialog.dismiss(); 
121          }

124      }
