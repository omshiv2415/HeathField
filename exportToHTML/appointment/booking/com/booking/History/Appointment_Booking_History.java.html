55       private class RemoteDataTask extends AsyncTask<Void, Void, Void> { 
56           @Override 
57           protected void onPreExecute() { 
58               super.onPreExecute(); 
59               // Create a progressdialog 
60               mProgressDialog = new ProgressDialog(Appointment_Booking_History.this); 
61               // Set progressdialog title 
62               mProgressDialog.setTitle("Please wait for Appointments"); 
63               // Set progressdialog message 
64               mProgressDialog.setMessage("Loading..."); 
65               mProgressDialog.setIndeterminate(false); 
66               // Show progressdialog 
67               mProgressDialog.show(); 
68           } 
69    
70           @Override 
71           protected Void doInBackground(Void... params) { 
72               // Create the array 
73               AppBookAndCancelList = new ArrayList<App_Booking_and_cancel_support>(); 
74               try { 
75                   // Locate the class table named "Country" in Parse.com 
76                   ParseQuery<ParseObject> query = new ParseQuery<ParseObject>( 
77                           "PatientBookingHistory"); 
78                   // Locate the column named "ranknum" in Parse.com and order list 
79                   // by ascending 
80                   query.orderByAscending("Doctor_Name"); 
81                   query.whereEqualTo("createdBy", ParseUser.getCurrentUser()); 
82                   ob = query.find(); 
83                   for (ParseObject country : ob) { 
84                       App_Booking_and_cancel_support map = new App_Booking_and_cancel_support(); 
85                       map.setDoctor_id((String) country.get("DrID")); 
86                       map.setDoctor_Name((String) country.get("DoctorName")); 
87                       map.setAppointment_Date(country.get("MyAppointment").toString()); 
88                       map.setAppointment_number((String) country.get("Appointment_No")); 
89                       map.setAppointment_Created(String.valueOf(country.getCreatedAt())); 
90                       AppBookAndCancelList.add(map); 
91                   } 
92               } catch (ParseException e) { 
93                   Log.e("Error", e.getMessage()); 
94                   e.printStackTrace(); 
95               } 
96               return null; 
97           } 
98    
99           @Override 
100          protected void onPostExecute(Void result) { 
101              TextView noApp = (TextView)findViewById(R.id.empty_history); 
102   
103   
104              Typeface Heading = Typeface.createFromAsset(getAssets(), "fonts/Sansation_Regular.ttf"); 
105              noApp.setTypeface(Heading); 
106              noApp.setTextSize(TypedValue.COMPLEX_UNIT_SP, 18); 
107   
108              // Locate the listview in activity_booking_appointmenting_appointment.xml 
109              listview = (ListView) findViewById(R.id.AppointmentListViewHistory); 
110              // Pass the results into Booking_ListViewAdapter.java 
111              adapter = new Cancel_History_ListViewAdapter(Appointment_Booking_History.this, 
112                      AppBookAndCancelList); 
113              // Binds the Adapter to the ListView 
114              if(adapter.getCount()!=0){ 
115                  noApp.setVisibility(View.INVISIBLE); 
116                  listview.setAdapter(adapter); 
117   
118              }else{ 
119                  Toast.makeText(Appointment_Booking_History.this, "No Appointments Booking History", Toast.LENGTH_SHORT).show(); 
120                  noApp.setVisibility(View.VISIBLE); 
121   
122              } 
123              // Close the progressdialog 
124              mProgressDialog.dismiss(); 
125          } 
126      }
