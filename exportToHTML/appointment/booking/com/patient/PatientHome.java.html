73                       @Override

75                           setProgressBarIndeterminateVisibility(false); 
76    
77                           if (e == null) { 
78                               //Success! 
79                               Toast.makeText(PatientHome.this, " Your Details updated successfully ", Toast.LENGTH_LONG).show(); 
80                               Intent intent = new Intent(PatientHome.this, PatientHome.class); 
81                               startActivity(intent); 
82    
83                           } else { 
84                               AlertDialog.Builder builder = new AlertDialog.Builder(PatientHome.this); 
85                               builder.setMessage(e.getMessage()) 
86                                       .setTitle(R.string.sign_up_error_title) 
87                                       .setPositiveButton(android.R.string.ok, null);
