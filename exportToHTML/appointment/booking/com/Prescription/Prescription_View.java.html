109                  if(pmedicatonrefill.equals("no")||pmedicatonrefill.equals("NO")) {

137                  // sending a notification to user once they booked the appointment 
138                      ParseQuery<ParseInstallation> pQuery = ParseInstallation.getQuery(); 
139                      pQuery.whereEqualTo("username", ParseUser.getCurrentUser()); 
140   
141                      ParsePush push = new ParsePush();

143                      push.setMessage("Confirmation of prescription order"); 
144                      push.sendInBackground();

146                      toast.setGravity(Gravity.CENTER, 0, 0);
