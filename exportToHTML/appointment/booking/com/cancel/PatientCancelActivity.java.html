82       private class RemoteDataTask extends AsyncTask<Void, Void, Void> { 
83           @Override 
84           protected void onPreExecute() { 
85               super.onPreExecute(); 
86               // Create a progressdialog 
87               mProgressDialog = new ProgressDialog(PatientCancelActivity.this); 
88               // Set progressdialog title 
89               mProgressDialog.setTitle("Please wait for Appointments"); 
90               // Set progressdialog message 
91               mProgressDialog.setMessage("Loading..."); 
92               mProgressDialog.setIndeterminate(false); 
93               // Show progressdialog 
94               mProgressDialog.show(); 
95           } 
96    
97           @Override 
98           protected Void doInBackground(Void... params) { 
99               // Create the array 
100              AppBookAndCancelList = new ArrayList<App_Booking_and_cancel_support>(); 
101              try { 
102                  // Locate the class table named "Country" in Parse.com 
103                  ParseQuery<ParseObject> query = new ParseQuery<ParseObject>( 
104                          "PatientAppointment"); 
105                  // Locate the column named "ranknum" in Parse.com and order list 
106                  // by ascending 
107                  query.orderByAscending("Doctor_Name"); 
108                  query.whereEqualTo("createdBy", ParseUser.getCurrentUser()); 
109                  ob = query.find(); 
110                  for (ParseObject country : ob) { 
111                      App_Booking_and_cancel_support map = new App_Booking_and_cancel_support(); 
112                      map.setDoctor_id((String) country.get("DrID")); 
113                      map.setDoctor_Name((String) country.get("DoctorName")); 
114                      map.setAppointment_Date((String) country.get("MyAppointment").toString()); 
115                      map.setAppointment_number((String) country.get("Appointment_No")); 
116                      AppBookAndCancelList.add(map); 
117                  } 
118              } catch (ParseException e) { 
119                  Log.e("Error", e.getMessage()); 
120                  e.printStackTrace(); 
121              } 
122              return null; 
123          } 
124   
125          @Override 
126          protected void onPostExecute(Void result) { 
127              // Locate the listview in activity_booking_appointmenting_appointment.xml 
128              TextView noApp = (TextView)findViewById(R.id.list_empty); 
129              listview = (ListView) findViewById(R.id.AppointmentListView); 
130              // Pass the results into Booking_ListViewAdapter.java 
131              adapter = new Cancel_ListViewAdapter(PatientCancelActivity.this, 
132                      AppBookAndCancelList); 
133              // Binds the Adapter to the ListView 
134              if(adapter.getCount()!=0){ 
135                  listview.setAdapter(adapter); 
136                  noApp.setVisibility(View.GONE); 
137              }else{ 
138                  Toast.makeText(PatientCancelActivity.this, "No Appointments To Cancel", Toast.LENGTH_SHORT).show(); 
139                  noApp.setVisibility(View.VISIBLE); 
140                  noApp.setText("No Appointments To Cancel"); 
141              } 
142              // Close the progressdialog 
143              mProgressDialog.dismiss(); 
144          } 
145      }
